CREATE DATABASE LAYOA;
USE LAYOA;
CREATE TABLE TB_ROLE(
	ROW_ID BIGINT PRIMARY KEY AUTO_INCREMENT COMMENT '主键',
	ROLE_CODE VARCHAR(30) NOT NULL COMMENT '角色编号',
	ROLE_NAME VARCHAR(30) NOT NULL UNIQUE COMMENT '角色名称#要求唯一',
	ROLE_KIND INT COMMENT '角色类型#1:一级角色;0:二级角色;',
	ROLE_INFO TEXT COMMENT '角色简介',
	ACTIVE_FLAG INT COMMENT '数据是否有效#1:是;0:否;',
	CREATE_BY VARCHAR(20) COMMENT '创建人',
	CREATE_DATE DATETIME COMMENT '创建时间',
	UPDATE_BY VARCHAR(20) COMMENT '修改人',
	UPDATE_DATE DATETIME COMMENT '修改时间'	
)COMMENT '角色表';
